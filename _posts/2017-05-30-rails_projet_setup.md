~~~
user@host:~$ gem install rails
user@host:~$ gem install bundler

user@host:~$ git clone my_project
user@host:~$ cd my_project

user@host:~$ bundle install
user@host:~$ rake db:create
user@host:~$ rake db:migrate
user@host:~$ rails s
~~~
